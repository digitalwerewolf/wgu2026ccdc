# ========================
# PHASE 2: PREFLIGHT — IDENTIFY INTERFACES (DO THIS FIRST)
# ========================
# Go to ESXi web UI > Select the firewall VM > Edit Settings
# Note which Network Adapter maps to which port group.
#
# OFFICIAL MAPPING (VM-Series Deployment Guide, page 169):
#   vNIC 1 (Network Adapter 1) = mgt (Ethernet 1/0) — NEVER TOUCH
#   vNIC 2 (Network Adapter 2) = ethernet1/1 (eth1) — outside data
#   vNIC 3 (Network Adapter 3) = ethernet1/2 (eth2) — inside data
#
# !! CHECK vSWITCH SECURITY SETTINGS (REQUIRED FOR VWIRE/L2) !!
# In ESXi: Networking > Virtual Switches
# For EACH port group connected to data interfaces (vNIC 2 and 3):
#   Edit Settings > Security tab > Set ALL THREE to Accept:
#     - Promiscuous Mode = Accept
#     - MAC Address Changes = Accept
#     - Forged Transmits = Accept
# WITHOUT THIS, THE FIREWALL RECEIVES ZERO DATA TRAFFIC.
# Do NOT change these on the management port group.
#
# Exception: L3 mode with hypervisor-assigned MAC (default) does NOT
# need promiscuous mode. Only vwire and L2 need it.
#
# !! NEVER DO THESE !!
# - NEVER run: set system setting mgmt-interface-swap enable yes
#   (This swaps mgmt to a data interface — instant lockout)
# - NEVER remove vNICs from the firewall VM (breaks interface mapping)
# - NEVER use VMware snapshots on the firewall VM (causes packet loss)
#   Use "Export named configuration snapshot" instead for backups.

# Then on the firewall CLI, confirm:

show interface management
# This shows the mgt IP — this is your lifeline, DO NOT TOUCH

debug show vm-series interfaces all
# This shows the ACTUAL vNIC-to-interface mapping with MAC addresses
# and PCI IDs. This is the definitive proof of which adapter is which.
# Example output:
#   Phoenix_interface  Base-OS_port  Base-OS_MAC           PCI-ID
#   mgt                eth0          0e:xx:xx:xx:xx:xx     0000:00:04.0
#   Ethernet1/1        eth1          0e:xx:xx:xx:xx:xx     0000:00:05.0
#   Ethernet1/2        eth2          0e:xx:xx:xx:xx:xx     0000:00:06.0

show interface all
# This shows ethernet1/1, ethernet1/2, etc. and their status
# Match these to the ESXi network adapters.
# Only the DATA interfaces (ethernet1/1, ethernet1/2) go in the vwire.
# If you see 3+ data interfaces, check the network diagram in the
# team packet to decide which is outside and which is inside.

# Verify no one has swapped the management interface:
# If the debug output above shows "mgt(interface-swap)" next to eth0,
# someone has enabled mgmt-interface-swap. This is dangerous.
# To undo: set system setting mgmt-interface-swap enable no
# Then: request restart system

# ========================
# PHASE 4b: BLOCK DANGEROUS PORTS
# ========================
# These are Red Team's favorite lateral movement/exploitation ports.
# This deny rule sits ABOVE the scored service allow rules.
# The Allow-Scoring rule (Phase 4) is above this, so the scoring
# engine still has full access. This blocks everyone ELSE.
#
# What's blocked:
#   SMB (445) — EternalBlue, PsExec, lateral movement, hash theft
#   RDP (3389) — brute force, BlueKeep, session hijacking
#   WinRM (5985/5986) — remote PowerShell execution
#   Telnet (23) — cleartext credentials, trivial exploitation
#   NetBIOS (137-139) — legacy name resolution abuse, enumeration
#   SNMP (161-162) — community string attacks, MIB enumeration
#
# If any of these turns out to be a scored service (very unlikely),
# remove that port from the block list.

configure

# Create service objects for each dangerous port
set service svc-block-tcp-445 protocol tcp port 445
set service svc-block-tcp-3389 protocol tcp port 3389
set service svc-block-tcp-5985 protocol tcp port 5985
set service svc-block-tcp-5986 protocol tcp port 5986
set service svc-block-tcp-23 protocol tcp port 23
set service svc-block-tcp-137 protocol tcp port 137
set service svc-block-tcp-138 protocol tcp port 138
set service svc-block-tcp-139 protocol tcp port 139
set service svc-block-udp-137 protocol udp port 137
set service svc-block-udp-138 protocol udp port 138
set service svc-block-udp-161 protocol udp port 161
set service svc-block-udp-162 protocol udp port 162

# Group them
set service-group Dangerous-Ports members [ svc-block-tcp-445 svc-block-tcp-3389 svc-block-tcp-5985 svc-block-tcp-5986 svc-block-tcp-23 svc-block-tcp-137 svc-block-tcp-138 svc-block-tcp-139 svc-block-udp-137 svc-block-udp-138 svc-block-udp-161 svc-block-udp-162 ]

# Create the deny rule with logging (both start and end for visibility)
set rulebase security rules Block-Dangerous-Inbound from any to any source any destination any application any service Dangerous-Ports action deny log-start yes log-end yes

# IMPORTANT: Move it above the scored service allow rules.
# If you used pa_deploy.py, it's already in the right order.
# If you created rules manually, move it now:
# move rulebase security rules Block-Dangerous-Inbound before <YOUR_FIRST_SCORED_SVC_RULE>

commit
